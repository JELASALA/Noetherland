Appendix 3.4C

Suhubi ABC Coefficients for N=1 (9.4.14)

\begin{equation}A = V\left( \Sigma  \right) + \frac{{\partial V\left( {{\Sigma ^i}} \right)}}{{\partial {x^i}}} + \Sigma \frac{{\partial {X^i}}}{{\partial {x^i}}} + \frac{{\partial {\Sigma ^i}}}{{\partial {x^i}}}\frac{{\partial {X^j}}}{{\partial {x^j}}} - \frac{{\partial {\Sigma ^i}}}{{\partial {x^j}}}\frac{{\partial {X^j}}}{{\partial {x^i}}}\tag{A}\end{equation}

\begin{equation}{A^i} = \frac{{\partial V\left( {{\Sigma ^i}} \right)}}{{\partial u}} + \Sigma \frac{{\partial {X^i}}}{{\partial u}} + \frac{{\partial {\Sigma ^j}}}{{\partial {x^j}}}\frac{{\partial {X^i}}}{{\partial u}} - \frac{{\partial {\Sigma ^i}}}{{\partial {x^j}}}\frac{{\partial {X^j}}}{{\partial u}} + \frac{{\partial {\Sigma ^i}}}{{\partial u}}\frac{{\partial {X^j}}}{{\partial {x^j}}} - \frac{{\partial {\Sigma ^j}}}{{\partial u}}\frac{{\partial {X^i}}}{{\partial {x^j}}}\tag{A1}\end{equation}

\begin{equation}{A^{ij}} = \frac{{\partial V\left( {{\Sigma ^i}} \right)}}{{\partial {v_j}}} + \Sigma \frac{{\partial {X^i}}}{{\partial {v_j}}} + \frac{{\partial {\Sigma ^k}}}{{\partial {x^k}}}\frac{{\partial {X^i}}}{{\partial {v_j}}} - \frac{{\partial {\Sigma ^i}}}{{\partial {x^k}}}\frac{{\partial {X^k}}}{{\partial {v_j}}} + \frac{{\partial {\Sigma ^i}}}{{\partial {v_j}}}\frac{{\partial {X^k}}}{{\partial {x^k}}} - \frac{{\partial {\Sigma ^k}}}{{\partial {v_j}}}\frac{{\partial {X^i}}}{{\partial {x^k}}}\tag{A2}\end{equation}

\begin{equation}{B^{ijk}} =  - {B^{jik}} = \frac{1}{2}\left[ {\frac{{\partial {\Sigma ^i}}}{{\partial {v_k}}}\frac{{\partial {X^j}}}{{\partial u}} - \frac{{\partial {\Sigma ^j}}}{{\partial {v_k}}}\frac{{\partial {X^i}}}{{\partial u}} + \frac{{\partial {\Sigma ^j}}}{{\partial u}}\frac{{\partial {X^i}}}{{\partial {v_k}}} - \frac{{\partial {\Sigma ^i}}}{{\partial u}}\frac{{\partial {X^j}}}{{\partial {v_k}}}} \right]A\tag{B3}\end{equation}

\begin{equation}{C^{ijkl}} =  - {C^{jikl}} =  - {C^{ijlk}} = \frac{1}{4}\left[ {\frac{{\partial {\Sigma ^i}}}{{\partial {v_l}}}\frac{{\partial {X^j}}}{{\partial {v_k}}} - \frac{{\partial {\Sigma ^j}}}{{\partial {v_l}}}\frac{{\partial {X^i}}}{{\partial {v_k}}} + \frac{{\partial {\Sigma ^j}}}{{\partial {v_k}}}\frac{{\partial {X^i}}}{{\partial {v_l}}} - \frac{{\partial {\Sigma ^i}}}{{\partial {v_k}}}\frac{{\partial {X^j}}}{{\partial {v_l}}}} \right]\tag{C4}\end{equation}

\begin{equation}{A^{i{i_1} \ldots {i_r}}} = \frac{{\partial V\left( {{\Sigma ^i}} \right)}}{{\partial {v_{{i_1} \ldots {i_r}}}}} + \frac{{\partial {\Sigma ^i}}}{{\partial {v_{{i_1} \ldots {i_r}}}}}\frac{{\partial {X^j}}}{{\partial {x^j}}} - \frac{{\partial {\Sigma ^j}}}{{\partial {v_{{i_1} \ldots {i_r}}}}}\frac{{\partial {X^i}}}{{\partial {x^j}}},\;\;\;2 \leqslant r \leqslant m\tag{Ar}\end{equation}

\begin{equation}{B^{ij{i_1} \ldots {i_r}}} =  - {B^{ji{i_1} \ldots {i_r}}} = \frac{1}{2}\left[ {\frac{{\partial {\Sigma ^i}}}{{\partial {v_{{i_1} \ldots {i_r}}}}}\frac{{\partial {X^j}}}{{\partial u}} - \frac{{\partial {\Sigma ^j}}}{{\partial {v_{{i_1} \ldots {i_r}}}}}\frac{{\partial {X^i}}}{{\partial u}}} \right],\;\;\;2 \leqslant r \leqslant m\tag{ Br }\end{equation}

\begin{equation}{C^{ijk{i_1} \ldots {i_r}}} =  - {C^{jik{i_1} \ldots {i_r}}} = \frac{1}{2}\left[ {\frac{{\partial {\Sigma ^i}}}{{\partial {v_{{i_1} \ldots {i_r}}}}}\frac{{\partial {X^j}}}{{\partial {v_k}}} - \frac{{\partial {\Sigma ^j}}}{{\partial {v_{{i_1} \ldots {i_r}}}}}\frac{{\partial {X^i}}}{{\partial {v_k}}}} \right],\;\;\;2 \leqslant r \leqslant m\tag{ Cr }\end{equation}
