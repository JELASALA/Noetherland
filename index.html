\begin{equation} {F^\alpha }\left( {{x^i},{u^\alpha },u_{,i}^\alpha ,u_{,ij}^\alpha , \ldots ,u_{,{i_1}{i_2} \ldots {i_m}}^\alpha } \right) = 0 \end{equation}

\begin{equation} 0 = {u_{xx}} - {u_t} = F\left( {x,t,{u_t},{u_{xx}}} \right) \to F\left( {{x^1},{x^2},{u_{{x^2}}},{u_{{x^1}{x^1}}}} \right) = 0\end{equation}
